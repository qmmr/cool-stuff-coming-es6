A brief history of the JavaScript

  • JavaScript was created in 1995 by Brendan Eich, an engineer at Netscape
  • first released with Netscape 2 early in 1996
  • originally called LiveScript to reflect its dynamic natue
  • renamed after the popular Sun Microsystem's Java language
  • despite the two having very little in common. This has been a source of confusion ever since.
  • Microsoft released a mostly-compatible version of the language called JScript with IE 3 several months later
  • Netscape submitted the language to Ecma International, a European standards organization, which resulted in the first edition of the ECMAScript standard in 1997.
  • The standard received a significant update as ECMAScript edition 3 in 1999, and has stayed pretty much stable ever since.
  • The fourth edition was abandoned, due to political differences concerning language complexity.
  • Many parts of the fourth edition formed a basis of the new ECMAScript edition 5, published in December of 2009.

old way

    name: 'Joe',

        setTimeout(function () {
            console.log( this.name )

    name: 'Joe',

...rest

is an array of params that were not declared

rest params - old way

sum function with 2 required params ( a, b ) any additional params are optional 

function sum ( a, b ) {
    var rest = [].slice.call(arguments, 2)
    var optional = 0

    if ( rest.length ) {
        optional = rest.reduce(function ( prev, current ) {
            return prev + current
        })
    }

    return a + b + optional
}
sum( 1, 2, 3, 4, 5 ) // 15

rest params - new way


function sum ( a, b, ...more ) {
    return a + b + more.reduce(function ( prev, current ) {
        return prev + current
    })
}
sum( 1, 2, 3, 4, 5 ) // 15

iterators

iterable objects have [Symbol.iterator] function that returns iterable object

currently no environment implements Symbol

some vendors (Nightly Firefox) have ["@@iterator"] property

iterator object has one method: next() which iterates over values of the object

iterating over array


let numbers = [ 1, 2 ]
let iter = numbers["@iterator"]() // iterator object that has method next()
iter.next() // { value: 1, done: false }
iter.next() // { value: 2, done: false }
iter.next() // { value: undefined, done: true }

generators

iterating over array


function* evenGen ( arr ) {
    for ( let num of arr ) {
        if ( num % 2 === 0 ) {
            yield num;
        }
    }
}
let numbers = [ 1, 2, 3, 4, 5, 6 ]
for ( let num of evenGen ( numbers ) ) {
    console.log( num )
}

iterating over obj using generator part 2


function* objGen ( obj ) {
    for ( let prop in obj ) {
        if (obj.hasOwnProperty(prop) ) {
            yield [prop, obj[prop]]
        }
    }
}

for ( let [key, val] of objGen( johnDoe ) ) {
    console.log( key + ': ' + val )
}
// "firstName: John"
// "lastName: Doe"

comprehensions

syntactic construct for creating arrays based on other arrays

ES5 - map


var jedis = ['Luke Skywalker', 'Obi-wan Kenobi', 'Yoda']
var jediMasters = jedis.map(function ( jedi ) {
    return 'master ' + jedi
})
console.log( jediMasters ) // [ "master Luke Skywalker", "master Obi-wan Kenobi", "master Yoda" ]

ES6 - map using comprehensions


let jedis = ['Luke Skywalker', 'Obi-wan Kenobi', 'Yoda']
let jediMasters = [ 'master ' + jedi for ( jedi of jedis ) ]

ES5 - filter


var jedis = [ 'Luke Skywalker', 'Yoda', 'Darth Vader', 'Darth Maul' ]
var jediKnights = jedis.filter(function ( jedi ) {
    return jedi.indexOf( 'Darth' ) == -1
})
console.log( jediKnights ) // [ 'Luke Skywalker', 'Yoda' ]

ES6 - filter using comprehensions


let jedis = [ 'Luke Skywalker', 'Yoda', 'Darth Vader', 'Darth Maul' ]
let jediKnights = [ knight for ( knight of jedis ) if ( knight.indexOf( 'Darth' ) == -1) ]
console.log( jediKnights )

Generator comprehension


let jedis = [ 'Luke Skywalker', 'Yoda', 'Darth Vader', 'Darth Maul' ]
let jediKnights = ( knight for ( knight of jedis ) if ( knight.indexOf( 'Darth' ) == -1) )
console.log( jediKnights.next() ) // 'Luke Skywalker'
