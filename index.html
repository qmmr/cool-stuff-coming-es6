New syntax

let

is the new var

Block-scoped binding


function printName ( language ) {
    var name = 'Marcin'
    if ( language === 'English' ) {
        let name = 'Martin'
        console.log( name ) // Martin
    }
}

Local scopes


function wrapper ( arr ) {
    var result = []
    for ( var i = 0, len = arr.length; i < len; ++i ) {
        result[i] = function () { return arr[i] }
    }
    return result
}
var wrapped = wrapper( ['Luke', 'Obi-Wan', 'Yoda'] )
wrapped[1] // undefined

Old solution


function wrapper ( arr ) {
    var result = []
    for ( var i = 0, len = arr.length; i < len; ++i ) {
        (function ( j ) {
            result[i] = function () { return arr[j] }
        })( i )
    }
    return result
}
var wrapped = wrapper( ['Luke', 'Obi-Wan', 'Yoda'] )
wrapped[2] // 'Yoda'

New solution


function wrapper ( arr ) {
    var result = []
    for ( let i = 0, len = arr.length; i < len; ++i ) {
        result[i] = function () { return arr[i] }
    }
    return result
}
var wrapped = wrapper( ['Luke', 'Obi-Wan', 'Yoda'] )
wrapped[1]() // 'Obi-Wan'

const

once declared, cannot be redeclared


const foo = 'hello world'
foo = 'oops!' // error

declaration must be followed by assignment

otherwise we have undefined value, that we cannot change (sic!)

IE will give an error, others will ignore it (Firefox)


const foo;
console.log( foo ) // undefined
foo = 'hello world!' // error: cannot reasign const

const (like let) are block scoped


if ( 1 ) {
    const foo = 'hello world!'
}
console.log( foo ) // undefined

const can hold any type of variables (array, object etc.)


const foo = []
foo.push( 'hello' )
foo.push( 'world!' )
console.log( foo.join(', ') ) // hello, world!
                        

const bar = {}
bar.baz = 'hello, world!'
console.log( bar.baz ) // hello, world!

default parameters

old way, checking if optional params were supplied


function createCustomElement ( tagName, id, parent ) {
    if ( !id ) {
        id = generateID()
    }
    if ( ! parent ) {
        parent = document.body
    }
    let element = document.createElement( tagName )
    element.id = id
    parent.appendChild( element )
    return element
}

new way, assignment inside parenthesis


function createCustomElement ( tagName, id = generateID(), parent = document.body ) {
    let element = document.createElement( tagName )
    element.id = id
    parent.appendChild( element )
    return element
}

...rest

is an array of params that were not declared

rest params - old way

sum function with 2 required params ( a, b )

any additional params are optional


function sum ( a, b ) {
    var rest = [].slice.call(arguments, 2)

    return a + b + rest.reduce(function ( prev, current ) {
        return prev + current
    })
}
sum( 1, 2, 3, 4, 5 ) // 15

rest params - new way


function sum ( a, b, ...more ) {
    return a + b + more.reduce(function ( prev, current ) {
        return prev + current
    })
}
sum( 1, 2, 3, 4, 5 ) // 15

...spread

expands array of args into individual variables

getting the highest value


let numbers = [ 0, 20, 21, 42 ]
// var highest = Math.max.apply(null, numbers) // 42
let highest = Math.max(...numbers) // 42

spread array into individual arguments


function coords ( x, y, z ) {
    return { x, y, z }
}
let points = [ 100, 50, 42 ]
// old way
var point = coords( points[0], points[1], points[2] )
// new way
let point = coords( ...points )
// or even shorter
let point = coords( ...[ 100, 50, 42 ] )
console.log(point.x, point.y, point.z) // 100 50 42

joining arrays - novice


// old way
var primes = [ 2, 3, 5 ]
var fibonacci = [ 0, 1, 1 ]
fibonacci = fibonacci.concat( primes ) // [0, 1, 1, 2, 3, 5]
// new way
fibonacci = [ 0, 1, 1, ...primes ]
console.log( fibonacci ) // [0, 1, 1, 2, 3, 5]

joining arrays - advanced


// old way
var primes = [ 2, 3, 5 ]
var fibonacci = [ 0, 1, 1, 8, 13, 21 ]
[ 2, 3, 5 ].forEach(function ( num ) {
    var index = fibonacci.indexOf( 8 )
    fibonacci.splice( index, 0, num )
})
// new way
let fibonacci = [ 0, 1, 1, ...primes, 8, 13, 21 ]
console.log( fibonacci ) // [0, 1, 1, 2, 3, 5, 8, 13, 21]

iterators

iterable objects have [Symbol.iterator] function that returns iterable object

currently no environment implements Symbol

some vendors (Nightly Firefox) have ["@@iterator"] property

iterator object has one method: next() which iterates over values of the object

iterating over array


let numbers = [ 1, 2 ]
let iter = numbers["@iterator"]() // iterator object that has method next()
iter.next() // { value: 1, done: false }
iter.next() // { value: 2, done: false }
iter.next() // { value: undefined, done: true }

generators

iterating over array


function* evenGen ( arr ) {
    for ( let num of arr ) {
        if ( num % 2 === 0 ) {
            yield num;
        }
    }
}
let numbers = [ 1, 2, 3, 4, 5, 6 ]
for ( let num of evenGen ( numbers ) ) {
    console.log( num )
}

iterating over obj using generator


function Person ( firstName, lastName ) {
    this.firstName = firstName
    this.lastName = lastName
}
Person.prototype.greet = function () {
    return 'Hi, my name is ' + this.firstName
}
let johnDoe = new Person( ...['John', 'Doe'] )
for ( let val of johnDoe ) {
    console.log( val )
}
// TypeError: john['@@iterator'] is not a function

function* objGen ( obj ) {
    for ( let prop in obj ) {
        if (obj.hasOwnProperty(prop) ) {
            yield [prop, obj[prop]]
        }
    }
}

for ( let [key, val] of objGen( johnDoe ) ) {
    console.log( key + ': ' + val )
}
// "firstName: John"
// "lastName: Doe"

comprehensions

syntactic construct for creating arrays based on other arrays

ES5 - map


var jedis = ['Luke Skywalker', 'Obi-wan Kenobi', 'Yoda']
var jediMasters = jedis.map(function ( jedi ) {
    return 'master ' + jedi
})
console.log( jediMasters ) // [ "master Luke Skywalker", "master Obi-wan Kenobi", "master Yoda" ]

ES6 - map using comprehensions


let jedis = ['Luke Skywalker', 'Obi-wan Kenobi', 'Yoda']
let jediMasters = [ 'master ' + jedi for ( jedi of jedis ) ]
