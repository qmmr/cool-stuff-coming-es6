New syntax

let

is the new var

Block-scoped binding


function printName ( language ) {
    var name = 'Marcin'
    if ( language === 'English' ) {
        let name = 'Martin'
        console.log( name ) // Martin
    }
}

Local scopes


function wrapper ( arr ) {
    var result = []
    for ( var i = 0, len = arr.length; i < len; ++i ) {
        result[i] = function () { return arr[i] }
    }
    return result
}
var wrapped = wrapper( ['Luke', 'Obi-Wan', 'Yoda'] )
wrapped[1] // undefined

Old solution


function wrapper ( arr ) {
    var result = []
    for ( var i = 0, len = arr.length; i < len; ++i ) {
        (function ( j ) {
            result[i] = function () { return arr[j] }
        })( i )
    }
    return result
}
var wrapped = wrapper( ['Luke', 'Obi-Wan', 'Yoda'] )
wrapped[2] // 'Yoda'

New solution


function wrapper ( arr ) {
    var result = []
    for ( let i = 0, len = arr.length; i < len; ++i ) {
        result[i] = function () { return arr[i] }
    }
    return result
}
var wrapped = wrapper( ['Luke', 'Obi-Wan', 'Yoda'] )
wrapped[1]() // 'Obi-Wan'

const

once declared, cannot be redeclared


const foo = 'hello world'
foo = 'oops!' // error

declaration must be followed by assignment

otherwise we have undefined value, that we cannot change (sic!)

IE will give an error, others will ignore it (Firefox)


const foo;
console.log( foo ) // undefined
foo = 'hello world!' // error: cannot reasign const

const (like let) are block scoped


if ( 1 ) {
    const foo = 'hello world!'
}
console.log( foo ) // undefined

const can hold any type of variables (array, object etc.)


const foo = []
foo.push( 'hello' )
foo.push( 'world!' )
console.log( foo.join(', ') ) // hello, world!
                        

const bar = {}
bar.baz = 'hello, world!'
console.log( bar.baz ) // hello, world!

default parameters

old way, checking if optional params were supplied


function createCustomElement ( tagName, id, parent ) {
    if ( !id ) {
        id = generateID()
    }
    if ( ! parent ) {
        parent = document.body
    }
    let element = document.createElement( tagName )
    element.id = id
    parent.appendChild( element )
    return element
}

new way, assignment inside parenthesis


function createCustomElement ( tagName, id = generateID(), parent = document.body ) {
    let element = document.createElement( tagName )
    element.id = id
    parent.appendChild( element )
    return element
}

rest operator

rest - is an array of params that were not declared

rest params - old way

sum function with 2 required params ( a, b )

any additional params are optional


function sum ( a, b ) {
    var rest = [].slice.call(arguments, 2)

    return a + b + rest.reduce(function ( prev, current ) {
        return prev + current
    })
}
sum( 1, 2, 3, 4, 5 ) // 15

rest params - new way


function sum ( a, b, ...more ) {
    return a + b + more.reduce(function ( prev, current ) {
        return prev + current
    })
}
sum( 1, 2, 3, 4, 5 ) // 15

spread operator

spread expands array of args into individual variables

getting the highest value


let numbers = [ 0, 20, 21, 42 ]
// var highest = Math.max.apply(null, numbers) // 42
let highest = Math.max(...numbers) // 42

spread array into individual arguments


function coords ( x, y, z ) {
    return { x, y, z }
}
let point = coords( ...[ 100, 50, 42 ] )
console.log(point.x, point.y, point.z) // 100 50 42
                        

let primes = [2, 3, 5]
let fibonacci = [0, 1, 1, ...primes, 8, 13, 21]
console.log(fibonacci) // [0, 1, 1, 2, 3, 5, 8, 13, 21]

destructuring

extract data from objects and arrays


// old way
let a = 1
let b = 2
let tmp = a
a = b
b = tmp
// new way
[a, b] = [b, a]
console.log(a, b) // 2 1
                        

function foo () {
    return [ 5, 10, 15 ]
}
let [ a, b, c ] = foo()
console.log(a, b, c) // 5 10 15
// omitting
let [ x, , z ] = foo()

long way / control naming


let person = {
    name: {
        first: 'John',
        last: 'Doe'
    }
}
let { first: firstName, last: lastName } = person.name
console.log(firstName, lastName) // John Doe

short way / same variables names


let person = {
    firstName: 'John',
    lastName: 'Doe'
}
let { firstName, lastName } = person
console.log(firstName, lastName) // John Doe

destructuring inside function parameters


let person = {
    age: 30,
    name: {
        first: 'John',
        last: 'Doe'
    }
}
function bar ( { age, name: { first: firstName, last: lastName } } ) {
    console.log(age, firstName, lastName)
}
bar(person) // 30 John Doe
